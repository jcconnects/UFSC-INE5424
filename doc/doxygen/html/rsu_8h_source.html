62inline RSU::RSU(unsigned int rsu_id, Unit unit, std::chrono::milliseconds period,
63 const void* data, unsigned int data_size)
64 : _rsu_id(rsu_id), _unit(unit), _period(period), _running(false),
65 _periodic_thread(this, &RSU::broadcast) {
66
67 db<RSU>(TRC) << "RSU::RSU() called with id=" << rsu_id << ", unit=" << unit << ", period=" << period.count() << "ms\n";
68
69 // Store broadcast data if provided
70 if (data && data_size > 0) {
71 _data.resize(data_size);
72 std::memcpy(_data.data(), data, data_size);
73 }
74
75 // Create network stack with RSU ID for MAC address generation
76 _network = new Network(_rsu_id);
77
78 // Create communicator using the network's protocol channel and RSU's address
79 Address rsu_addr(_network->address(), _rsu_id);
80 _comm = new Communicator(_network->channel(), rsu_addr);
81
82 db<RSU>(INF) << "[RSU] RSU " << _rsu_id << " initialized with address " << rsu_addr.to_string() << "\n";
83
84 // Set self ID for the Clock instance
85 LeaderIdType self_leader_id = static_cast<LeaderIdType>(rsu_addr.paddr().bytes[5]);
86 if (self_leader_id != INVALID_LEADER_ID) {
87 Clock::getInstance().setSelfId(self_leader_id);
88 db<RSU>(INF) << "[RSU] RSU " << _rsu_id << " registered self_id " << self_leader_id << " with Clock.\n";
89 Clock::getInstance().activate(nullptr); // Activate clock to evaluate leader state
90 } else {
91 db<RSU>(WRN) << "[RSU] RSU " << _rsu_id << " has an INVALID_LEADER_ID based on its MAC. Clock self_id not set.\n";
92 }
93}

125inline void RSU::stop() {
126 db<RSU>(TRC) << "RSU::stop() called!\n";
127
128 if (_running.load(std::memory_order_acquire)) {
129 // Step 1: Signal RSU that it should stop its operations
130 _running.store(false, std::memory_order_release);
131 db<RSU>(INF) << "[RSU] RSU " << _rsu_id << " stopping broadcasting\n";
132
133 // Step 2: Stop periodic thread and wait for it to finish.
134 // This ensures RSU::broadcast() (and thus _comm->send()) is no longer called.
135 _periodic_thread.join();
136 db<RSU>(INF) << "[RSU] RSU " << _rsu_id << " periodic thread stopped\n";
137
138 // Step 3: Release communicator now that the thread is guaranteed to not use it.
139 _comm->release();
140
141 // Step 4: Stop network stack after threads are fully stopped
142 _network->stop();
143 db<RSU>(INF) << "[RSU] RSU " << _rsu_id << " stopped broadcasting\n";
144 }
145}

197inline void RSU::broadcast() {
198 if (!running()) {
199 return;
200 }
201
202 db<RSU>(TRC) << "RSU::broadcast() called!\n";
203
204 // Create RESPONSE message
205 Message* msg;
206 if (_data.empty()) {
207 // No data payload
208 msg = new Message(Message::Type::RESPONSE, address(), _unit);
209 } else {
210 // With data payload
211 msg = new Message(Message::Type::RESPONSE, address(), _unit,
212 Message::ZERO, _data.data(), _data.size());
213 }
214
215 db<RSU>(TRC) << "[RSU] RSU " << _rsu_id << " broadcasting RESPONSE for unit " << _unit
216 << " with data size " << _data.size() << "\n";
217
218 // Send broadcast message
219 bool sent = _comm->send(msg);
220
221 if (sent) {
222 db<RSU>(INF) << "[RSU] RSU " << _rsu_id << " broadcast RESPONSE for unit " << _unit << "\n";
223 } else {
224 db<RSU>(WRN) << "[RSU] RSU " << _rsu_id << " failed to broadcast RESPONSE for unit " << _unit << "\n";
225 }
226
227 delete msg;
228}
