55inline Vehicle::Vehicle(unsigned int id) : _id(id), _running(false)
56{
57 _gateway = new Gateway(_id);
58
59 // Set up CSV logging directory
60 _log_dir = CSVLogger::create_vehicle_log_dir(_id);
61 setup_csv_logging();
62
63 // Set self ID for the Clock instance
64 // Assuming Gateway's address is the Vehicle's primary address for PTP ID purposes
65 // The last byte of the MAC address is used as LeaderIdType
66 LeaderIdType self_leader_id = static_cast<LeaderIdType>(_gateway->address().paddr().bytes[5]);
67 if (self_leader_id != INVALID_LEADER_ID) {
68 Clock::getInstance().setSelfId(self_leader_id);
69 db<Vehicle>(INF) << "[Vehicle " << _id << "] registered self_id " << self_leader_id << " with Clock.\n";
70 Clock::getInstance().activate(nullptr); // Activate clock to evaluate leader state
71 } else {
72 db<Vehicle>(WRN) << "[Vehicle " << _id << "] has an INVALID_LEADER_ID based on its Gateway MAC. Clock self_id not set.\n";
73 }
74
75 // HARDCODE: Set RSU as the leader in LeaderKeyStorage for vehicles
76 // This matches the RSU_ID from the demo (1000)
77 const unsigned int RSU_ID = 1000;
78 Ethernet::Address rsu_mac;
79 rsu_mac.bytes[0] = 0x02; // Locally administered (matches demo.cpp setup)
80 rsu_mac.bytes[1] = 0x00;
81 rsu_mac.bytes[2] = 0x00;
82 rsu_mac.bytes[3] = 0x00;
83 rsu_mac.bytes[4] = (RSU_ID >> 8) & 0xFF; // = 3 for RSU_ID 1000
84 rsu_mac.bytes[5] = RSU_ID & 0xFF; // = 232 for RSU_ID 1000
85
86 auto& storage = LeaderKeyStorage::getInstance();
87 storage.setLeaderId(rsu_mac);
88
89 // Set a dummy MAC key for completeness (matches demo.cpp)
90 MacKeyType rsu_key;
91 rsu_key.fill(0);
92 rsu_key[0] = (RSU_ID >> 8) & 0xFF;
93 rsu_key[1] = RSU_ID & 0xFF;
94 rsu_key[2] = 0xAA; // Marker for RSU
95 rsu_key[3] = 0xBB;
96 storage.setGroupMacKey(rsu_key);
97
98 db<Vehicle>(INF) << "[Vehicle " << _id << "] set RSU " << RSU_ID << " as leader with MAC: "
99 << Ethernet::mac_to_string(rsu_mac) << " (leader_id=" << static_cast<unsigned int>(rsu_mac.bytes[5]) << ")\n";
100}

153
