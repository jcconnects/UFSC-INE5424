98inline Vehicle::Vehicle(unsigned int id) : _id(id), _running(false)
99{
100 _gateway = new Gateway(_id, Network::EntityType::VEHICLE);
101 _rsu_manager = std::make_unique<VehicleRSUManager<Gateway::Protocol>>(_id);
102 _gateway->network()->set_vehicle_rsu_manager(_rsu_manager.get());
103
104 // Set up CSV logging directory
105 _log_dir = CSVLogger::create_vehicle_log_dir(_id);
106 setup_csv_logging();
107
108 // Set self ID for the Clock instance (will be updated by RSU manager when leader is selected)
109 LeaderIdType self_leader_id = static_cast<LeaderIdType>(_gateway->address().paddr().bytes[5]);
110 if (self_leader_id != INVALID_LEADER_ID) {
111 Clock::getInstance().setSelfId(self_leader_id);
112 db<Vehicle>(INF) << "[Vehicle " << _id << "] registered self_id " << self_leader_id << " with Clock.\n";
113 Clock::getInstance().activate(nullptr);
114 }
115 db<Vehicle>(INF) << "[Vehicle " << _id << "] initialized with RSU management\n";
116}

118inline Vehicle::~Vehicle() {
119 // Ensure vehicle is stopped first
120 if (running()) {
121 stop();
122 }
123
124 // Components are managed by unique_ptr, destruction is automatic.
125 // But we clear them explicitly to ensure proper destruction order
126 _components.clear();
127
128 // Protocol and NIC are owned by Vehicle in this design
129 db<Vehicle>(TRC) << "Vehicle::~Vehicle() called for ID " << _id << "!\n";
130 delete _gateway;
131 db<Vehicle>(INF) << "[Vehicle " << _id << "] destroyed successfully.\n";
132}

134inline void Vehicle::start() {
135 db<Vehicle>(TRC) << "Vehicle::start() called for ID " << _id << "!\n";
136 if (running()) {
137 db<Vehicle>(WRN) << "[Vehicle " << _id << "] start() called but already running.\n";
138 return;
139 }
140
141 _running.store(true, std::memory_order_release);
142 _gateway->start();
143
144 db<Vehicle>(INF) << "[Vehicle " << _id << "] started.\n";
145}
146
147inline void Vehicle::stop() {
148 db<Vehicle>(TRC) << "Vehicle::stop() called for ID " << _id << "!\n";
149
150 if (!running()) {
151 db<Vehicle>(WRN) << "[Vehicle " << _id << "] stop() called but not running.\n";
152 return;
153 }
154
155 _running.store(false, std::memory_order_release); // Mark vehicle as stopped
156 db<Vehicle>(INF) << "[Vehicle " << _id << "] stopped.\n";
157}
