33
34 // Check if CSV file is loaded
35 if (!csv_data->is_loaded()) {
36 db<void>(WRN) << "[CSVComponent] No CSV file loaded or no data available\n";

38 }
39
40 // Get the next CSV record
41 const CSVComponentData::CSVRecord* record = csv_data->get_next_record();
42 if (!record) {
43 db<void>(ERR) << "[CSVComponent] Failed to get next CSV record\n";
44 return std::vector<std::uint8_t>();
45 }
46
47 // Serialize CSV record to byte vector with timestamp prepended
48 std::vector<std::uint8_t> result(sizeof(std::uint64_t) + sizeof(CSVComponentData::CSVRecord));
49
50 // Copy timestamp to the beginning of the result vector
51 std::memcpy(result.data(), &record->timestamp, sizeof(std::uint64_t));
52
53 // Copy the entire CSV record after the timestamp
54 std::memcpy(result.data() + sizeof(std::uint64_t), record, sizeof(CSVComponentData::CSVRecord));
55
56 // Log CSV data transmission
57 db<void>(INF) << "[CSVComponent] Sending CSV record #" << csv_data->get_records_sent()

60 << " id: " << record->id
61 << " pos: (" << record->x << ", " << record->y << ", " << record->z << ")"

64
65 // Additional detailed logging for debugging (key field values)
