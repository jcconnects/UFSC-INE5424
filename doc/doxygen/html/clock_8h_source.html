175inline void Clock::activate(const PtpRelevantData* new_msg_data) {
176 std::lock_guard<std::mutex> lock(_mutex);
177
178 // Get current leader from storage without holding its mutex
179 LeaderIdType storage_leader_id = INVALID_LEADER_ID;
180 {
181 auto& storage = LeaderKeyStorage::getInstance();
182 Ethernet::Address storage_leader = storage.getLeaderId();
183 storage_leader_id = storage_leader.bytes[5];
184 }
185
186 // NEW: Check if this node IS the leader (according to storage)
187 if (_self_id != INVALID_LEADER_ID && _self_id == storage_leader_id) {
188 if (_currentState.load(std::memory_order_acquire) != State::SYNCHRONIZED || _current_leader_id != _self_id) {
189 db<Clock>(INF) << "Clock: This node (" << _self_id << ") is the PTP leader. Forcing SYNCHRONIZED state.\n";
190 _currentState.store(State::SYNCHRONIZED, std::memory_order_release);
191 _current_leader_id = _self_id; // This node is the leader
192 _current_offset = DurationType::zero();
193 _current_drift_fe = 0.0;
194 _local_time_at_last_sync_event = getLocalSteadyHardwareTime();
195 _leader_time_at_last_sync_event = _local_time_at_last_sync_event; // Leader time is its own local time
196 _msg1_data = PtpInternalMessageInfo(); // Clear old sync message data
197 _msg2_data = PtpInternalMessageInfo();
198 _leader_has_changed_flag.store(false, std::memory_order_release);
199 }
200 // As a leader, its clock is authoritative. No further processing of PTP messages for sync.
201 return;
202 }
203
204 // If this node is NOT the leader, proceed with normal PTP logic.
205 // Check and handle leader change (original logic, might set state to UNSYNCHRONIZED)
206 [[maybe_unused]] bool actual_leader_change_occurred = checkAndHandleLeaderChange(storage_leader_id);
207
208 State current_state_local = _currentState.load(std::memory_order_acquire);
209 State new_state = current_state_local;
210
211 // Main state transitions
212 switch (current_state_local) {
213 case State::UNSYNCHRONIZED:
214 if (new_msg_data && isLeaderAssigned() && isMessageFromCurrentLeader(*new_msg_data)) {
215 new_state = State::AWAITING_SECOND_MSG;
216 doProcessFirstLeaderMsg(*new_msg_data);
217 }
218 break;
219
220 case State::AWAITING_SECOND_MSG:
221 if (isLeaderMessageTimedOut()) {
222 new_state = State::UNSYNCHRONIZED;
223 doClearSyncData();
224 } else if (new_msg_data && isMessageFromCurrentLeader(*new_msg_data)) {
225 new_state = State::SYNCHRONIZED;
226 doProcessSecondLeaderMsgAndCalcDrift(*new_msg_data);
227 }
228 break;
229
230 case State::SYNCHRONIZED:
231 if (isLeaderMessageTimedOut()) {
232 new_state = State::UNSYNCHRONIZED;
233 doClearSyncData();
234 } else if (new_msg_data && isMessageFromCurrentLeader(*new_msg_data)) {
235 doProcessSubsequentLeaderMsg(*new_msg_data);
236 }
237 break;
238 }
239
240 // Log state transition if it changed
241 if (new_state != current_state_local) {
242 db<Clock>(INF) << "Clock: " << current_state_local << " -> " << new_state << "\n";
243 _currentState.store(new_state, std::memory_order_release);
244 }
245}

253inline TimestampType Clock::getSynchronizedTime(bool* is_synchronized) {
254 std::lock_guard<std::mutex> lock(_mutex);
255 TimestampType local_hw_now = getLocalSteadyHardwareTime();
256 State current_state_local = _currentState.load(std::memory_order_acquire);
257
258 if (current_state_local == State::UNSYNCHRONIZED) {
259 db<Clock>(INF) << "Clock::getSynchronizedTime WARNING: Clock UNSYNCHRONIZED. Returning local hardware time.\n";
260 *is_synchronized = false;
261 return local_hw_now;
262 }
263
264 DurationType elapsed_since_last_sync = local_hw_now - _local_time_at_last_sync_event;
265
266 if (current_state_local == State::AWAITING_SECOND_MSG) {
267 // Only offset correction: SynchronizedTime = LeaderTimeAtSync + ElapsedLocal
268 // Note: _current_offset = _local_time_at_last_sync_event - _leader_time_at_last_sync_event
269 // So, LeaderTimeAtSync = _local_time_at_last_sync_event - _current_offset
270 // SynchronizedTime = (_local_time_at_last_sync_event - _current_offset) + elapsed_since_last_sync
271 // SynchronizedTime = local_hw_now - _current_offset
272 *is_synchronized = false;
273 return local_hw_now - _current_offset;
274 }
275
276 // State::SYNCHRONIZED: Apply offset and drift correction
277 // SynchronizedTime = local_hw_now - (_current_offset_at_last_event + drift_correction_for_elapsed_time)
278 // Or, using the state machine's variables:
279 // leader_increment = elapsed_local * (1.0 - current_drift_fe)
280 // sync_time = leader_time_at_last_sync_event + leader_increment
281 double leader_increment_double = static_cast<double>(elapsed_since_last_sync.count()) * (1.0 - _current_drift_fe);
282 DurationType leader_increment = std::chrono::microseconds(static_cast<long long>(leader_increment_double));
283 *is_synchronized = true;
284 return _leader_time_at_last_sync_event + leader_increment;
285}

304inline Clock::State Clock::getState() {
305 std::lock_guard<std::mutex> lock(_mutex);
306
307 State current_state_local = _currentState.load(std::memory_order_acquire);
308
309 // NEW: If this node is the leader, it's always synchronized.
310 if (_self_id != INVALID_LEADER_ID && _current_leader_id == _self_id) {
311 if (current_state_local != State::SYNCHRONIZED) {
312 // This ensures that if somehow the state was different, it's corrected.
313 db<Clock>(INF) << "Clock::getState: This node (" << _self_id << ") is leader. Correcting state to SYNCHRONIZED.\n";
314 _currentState.store(State::SYNCHRONIZED, std::memory_order_release);
315 }
316 return State::SYNCHRONIZED;
317 }
318
319 // Only check timeout for states that expect leader messages (non-leader case)
320 if (current_state_local == State::AWAITING_SECOND_MSG ||
321 current_state_local == State::SYNCHRONIZED) {
322
323 if (isLeaderMessageTimedOut()) {
324 db<Clock>(INF) << "Clock: Timeout detected, transitioning to UNSYNCHRONIZED\n";
325 _currentState.store(State::UNSYNCHRONIZED, std::memory_order_release);
326 doClearSyncData();
327 return State::UNSYNCHRONIZED;
328 }
329 }
330 return current_state_local;
331}

386inline void Clock::doProcessFirstLeaderMsg(const PtpRelevantData& msg_data) {
387 // Action for: UNSYNCHRONIZED --> AWAITING_SECOND_MSG
388 _msg1_data.ts_tx_at_sender = msg_data.ts_tx_at_sender;
389 _msg1_data.ts_local_rx = msg_data.ts_local_rx;
390
391 _msg1_data.d_tx_calc = std::chrono::microseconds(2);
392
393 _msg1_data.leader_time_at_local_rx_event = _msg1_data.ts_tx_at_sender + _msg1_data.d_tx_calc;
394 _current_offset = _msg1_data.ts_local_rx - _msg1_data.leader_time_at_local_rx_event;
395
396 _leader_time_at_last_sync_event = _msg1_data.leader_time_at_local_rx_event;
397 _local_time_at_last_sync_event = _msg1_data.ts_local_rx;
398 _current_drift_fe = 0.0;
399
400 db<Clock>(INF) << "Clock: Processed first leader message. Offset: "
401 << _current_offset.count() << "us\n";
402}

404inline void Clock::doProcessSecondLeaderMsgAndCalcDrift(const PtpRelevantData& msg_data) {
405 // Action for: AWAITING_SECOND_MSG --> SYNCHRONIZED
406 _msg2_data.ts_tx_at_sender = msg_data.ts_tx_at_sender;
407 _msg2_data.ts_local_rx = msg_data.ts_local_rx;
408
409 _msg2_data.d_tx_calc = std::chrono::microseconds(2);
410 _msg2_data.leader_time_at_local_rx_event = _msg2_data.ts_tx_at_sender + _msg2_data.d_tx_calc;
411
412 DurationType o1 = _msg1_data.ts_local_rx - _msg1_data.leader_time_at_local_rx_event;
413 DurationType o2 = _msg2_data.ts_local_rx - _msg2_data.leader_time_at_local_rx_event;
414 _current_offset = o2;
415
416 DurationType delta_o = o2 - o1;
417 // Use effective leader timestamps at local reception events for delta_T
418 DurationType delta_t_leader_effective = _msg2_data.leader_time_at_local_rx_event - _msg1_data.leader_time_at_local_rx_event;
419
420 if (delta_t_leader_effective.count() > 0) {
421 _current_drift_fe = static_cast<FrequencyErrorType>(delta_o.count()) / static_cast<FrequencyErrorType>(delta_t_leader_effective.count());
422 } else {
423 // std::cerr << "Clock WARNING: delta_T_leader_effective is zero or negative during drift calculation. Drift not updated from " << _current_drift_fe << ".\n";
424 }
425
426 _leader_time_at_last_sync_event = _msg2_data.leader_time_at_local_rx_event;
427 _local_time_at_last_sync_event = _msg2_data.ts_local_rx;
428
429 db<Clock>(INF) << "Clock: Processed second leader message. New Offset: "
430 << _current_offset.count() << "us, Drift FE: " << _current_drift_fe << "\n";
431}

442inline bool Clock::isLeaderMessageTimedOut() const {
443 // NEW: If this node is the current leader, it doesn't time out on itself.
444 // Note: _current_leader_id and _self_id are read. Callers must hold _mutex.
445 if (_self_id != INVALID_LEADER_ID && _current_leader_id == _self_id) {
446 return false;
447 }
448
449 if (_local_time_at_last_sync_event == TimestampType::min() || !isLeaderAssigned()) {
450 // No sync event yet or no leader, so not "timed out" in the sense of expecting a message
451 return false;
452 }
453 TimestampType local_hw_now = getLocalSteadyHardwareTime();
454 return (local_hw_now - _local_time_at_last_sync_event) > MAX_LEADER_SILENCE_INTERVAL;
455}

477inline bool Clock::checkAndHandleLeaderChange(LeaderIdType storage_leader_id) {
478 if (_current_leader_id != storage_leader_id) {
479 db<Clock>(INF) << "Clock: Leader changed from " << _current_leader_id
480 << " to " << storage_leader_id << " during activation\n";
