214template <typename NIC>
215int Protocol<NIC>::send(Address from, Address to, const void* data, unsigned int size) {
216 db<Protocol>(TRC) << "Protocol<NIC>::send() called!\n";
217
218 if (!_nic) {
219 db<Protocol>(TRC) << "Protocol<NIC>::send() called after release!\n";
220 return 0;
221 }
222
223 // Allocate buffer for the entire frame -> NIC alloc adds Frame Header size (this is better for independency)
224 unsigned int packet_size = size + sizeof(Header) + sizeof(TimestampFields);
225 Buffer* buf = _nic->alloc(to.paddr(), PROTO, packet_size);
226
227 // Get pointer to where Protocol packet starts (within the Ethernet payload)
228 // Assumes buf->data() returns Ethernet::Frame* and payload follows header
229 Packet* packet = reinterpret_cast<Packet*>(buf->data()->payload);
230
231 // Set up Protocol Packet Header
232 packet->from_port(from.port());
233 packet->to_port(to.port());
234 packet->size(size); // Set the size of the *user data*
235
236 // Copy user data into the packet's data section
237 std::memcpy(packet->template data<void>(), data, size);
238
239 // Set clock synchronization status in the packet
240 auto& clock = Clock::getInstance();
241 bool sync_status;
242 clock.getSynchronizedTime(&sync_status); // We only need the status
243 packet->timestamps()->is_clock_synchronized = sync_status;
244
245 // Send the packet via NIC, passing the packet size for timestamp offset calculation
246 int result = _nic->send(buf, packet_size); // Pass packet size to NIC
247 db<Protocol>(INF) << "[Protocol] NIC::send() returned " << result
248 << ", clock_synchronized=" << packet->timestamps()->is_clock_synchronized << "\n";
249
250 // NIC should release buffer after use
251 return result;
252}

254template <typename NIC>
255int Protocol<NIC>::receive(Buffer* buf, Address *from, void* data, unsigned int size) {
256 db<Protocol>(TRC) << "Protocol<NIC>::receive() called!\n";
257
258 typename NIC::Address src_mac;
259 typename NIC::Address dst_mac;
260
261 std::uint8_t temp_buffer[size + sizeof(Header) + sizeof(TimestampFields)];
262
263 if (!_nic) {
264 db<Protocol>(TRC) << "Protocol<NIC>::receive() called after release!\n";
265 return -1;
266 }
267
268 int packet_size = _nic->receive(buf, &src_mac, &dst_mac, temp_buffer, NIC::MTU);
269 db<Protocol>(INF) << "[Protocol] NIC::receive() returned " << packet_size << ".\n";
270
271 if (packet_size <= 0) {
272 db<Protocol>(WRN) << "[Protocol] No data received or error occurred.\n";
273 return -1; // No data received or error
274 }
275
276 // Reinterpretation as packet
277 Packet* pkt = reinterpret_cast<Packet*>(temp_buffer);
278
279 if (from) {
280 from->paddr(src_mac);
281 from->port(pkt->header()->from_port());
282 }
283
284 // Payload size
285 int payload_size = packet_size - sizeof(Header) - sizeof(TimestampFields);
286
287 // Copies only useful data
288 std::memcpy(data, pkt->template data<void>(), payload_size);
289
290 db<Protocol>(INF) << "[Protocol] received packet from " << Ethernet::mac_to_string(src_mac) << " to " << Ethernet::mac_to_string(dst_mac) << " with size " << packet_size << "\n";
291
292 return payload_size;
293}

307template <typename NIC>
308void Protocol<NIC>::update(typename NIC::Protocol_Number prot, Buffer * buf) {
309 db<Protocol>(TRC) << "Protocol<NIC>::update() called!\n";
310
311 if (!buf) {
312 db<Protocol>(INF) << "[Protocol] data received, but buffer is null. Releasing buffer.\n";
313 return;
314 }
315
316 Packet* pkt = reinterpret_cast<Packet*>(buf->data()->payload);
317
318 // Extract timestamps and update Clock if this is a PTP-relevant message
319 TimestampFields* timestamps = pkt->timestamps();
320
321 // Log the synchronization status from the sender
322 db<Protocol>(INF) << "[Protocol] Received packet with sender_clock_synchronized="
323 << timestamps->is_clock_synchronized << "\n";
324
325 // convert buffer (std::int64_t) _rx_time back to TimestampType
326 TimestampType rx_timestamp(std::chrono::milliseconds(buf->rx()));
327
328 typename NIC::Address src_mac = buf->data()->src;
329
330 // Create PTP data structure for Clock
331 PtpRelevantData ptp_data;
332 ptp_data.sender_id = static_cast<LeaderIdType>(src_mac.bytes[5]); // Use last byte of MAC as sender ID
333 ptp_data.ts_tx_at_sender = timestamps->tx_timestamp;
334 ptp_data.ts_local_rx = rx_timestamp;
335
336 // Update Clock with timing information
337 auto& clock = Clock::getInstance();
338 clock.activate(&ptp_data);
339
340 if (!Protocol::_observed.notify(buf)) { // Notify every listener
341 db<Protocol>(INF) << "[Protocol] data received, but no one was notified for port. Releasing buffer.\n";
342 free(buf);
343 }
344 db<Protocol>(INF) << "[Protocol] data received, notify succeeded.\n";
345}
