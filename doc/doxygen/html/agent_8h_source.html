67inline Agent::Agent(CAN* bus, const std::string& name, Unit unit, Type type, Address address) : _address(address), _name(name), _periodic_thread(nullptr) {
68 db<Agent>(INF) << "[Agent] " << _name << " created with address: " << _address.to_string() << "\n";
69 if (!bus)
70 throw std::invalid_argument("Gateway cannot be null");
71
72 _can = bus;
73 Condition c(unit, type);
74 _c = c;
75 _can_observer = new Observer(c);
76 _can->attach(_can_observer, c);
77
78 // Only send initial INTEREST if this agent is a consumer (observing RESPONSE messages)
79 // Producers (observing INTEREST messages) should not send INTEREST
80 if (type == Type::RESPONSE) {
81 Microseconds period(1000000);
82 send(unit, period); // Send initial INTEREST
83 }
84
85 _running = true;
86 int result = pthread_create(&_thread, nullptr, Agent::run, this);
87 if (result != 0) {
88 _running = false;
89 throw std::runtime_error("Failed to create agent thread");
90 }
91}

93inline Agent::~Agent() {
94 // First, stop the running flag to prevent new operations
95 _running.store(false, std::memory_order_release);
96
97 // Stop periodic thread first if it exists - CRITICAL: Do this before anything else
98 if (_periodic_thread) {
99 _periodic_thread->join();
100 delete _periodic_thread;
101 _periodic_thread = nullptr;
102 }
103
104 // Send a dummy message to wake up the main thread if it's waiting
105 Message* dummy_msg = new Message();
106 _can_observer->update(_c, dummy_msg);
107
108 // Wait for the main thread to finish
109 pthread_join(_thread, nullptr);
110
111 // Detach from CAN bus before deleting observer
112 if (_can_observer) {
113 _can->detach(_can_observer, _c);
114 delete _can_observer;
115 _can_observer = nullptr;
116 }
117
118 // Clean up the dummy message
119 delete dummy_msg;
120
121 db<Agent>(INF) << "[Agent] " << _name << " destroyed successfully\n";
122}

153inline void* Agent::run(void* arg) {
154 Agent* agent = reinterpret_cast<Agent*>(arg);
155
156 while (agent->running()) {
157 Message* msg = new Message();
158 int received = agent->receive(msg);
159
160 if (received <= 0 || !msg) {
161 db<Agent>(WRN) << "[Agent] " << agent->name() << " received an empty (received=" << received << ") or invalid message\n";
162 delete msg; // Clean up the message object
163
164 continue; // Skip processing if no valid message was received
165 }
166
167 db<Agent>(INF) << "[Agent] " << agent->name() << " received message of type: " << static_cast<int>(msg->message_type())
168 << " for unit: " << msg->unit() << " with size: " << msg->value_size() << "\n";
169
170 if (msg->message_type() == Message::Type::RESPONSE)
171 agent->handle_response(msg);
172 else if (msg->message_type() == Message::Type::INTEREST)
173 agent->handle_interest(msg->unit(), msg->period());
174
175 delete msg; // Clean up the message object after processing
176 }
177
178 return nullptr;
179}

185inline void Agent::handle_interest(Unit unit, Microseconds period) {
186 db<Agent>(INF) << "[Agent] " << _name << " received INTEREST for unit: " << unit << " with period: " << period.count() << " microseconds\n";
187
188 // Only respond to INTEREST if this agent is a producer (observing INTEREST messages)
189 // Consumers (observing RESPONSE messages) should not respond to INTEREST
190 if (_c.type() != Type::INTEREST) {
191 db<Agent>(WRN) << "[Agent] " << _name << " ignoring INTEREST message (not a producer)\n";
192 return;
193 }
194
195 if (!_periodic_thread) {
196 _periodic_thread = new Thread(this, &Agent::reply, unit);
197 _periodic_thread->start(period.count()); // Actually start the thread!
198 } else {
199 // Ajusta o período com MDC entre o atual e o novo
200 _periodic_thread->adjust_period(period.count());
201 }
202}

205inline void Agent::reply(Unit unit) {
206 // Safety check: don't reply if agent is being destroyed
207 if (!running()) {
208 return;
209 }
210
211 // Additional safety check: ensure periodic thread is still valid
212 if (!_periodic_thread || !_periodic_thread->running()) {
213 return;
214 }
215
216 db<Agent>(INF) << "[Agent] " << _name << " sending RESPONSE for unit: " << unit << "\n";
217
218 // Final safety check before calling virtual method
219 if (!running()) {
220 return;
221 }
222
223 Value value = get(unit);
224 Message msg(Message::Type::RESPONSE, _address, unit, Microseconds::zero(), value.data(), value.size());
225
226 // Log sent message to CSV
227 log_message(msg, "SEND");
228
229 _can->send(&msg);
230}

238inline void Agent::log_message(const Message& msg, const std::string& direction) {
239 if (!_csv_logger || !_csv_logger->is_open()) return;
240
241 auto timestamp_us = Message::getSynchronizedTimestamp().count();
242
243 // Calculate latency for received messages
244 auto latency_us = 0L;
245 if (direction == "RECEIVE") {
246 latency_us = timestamp_us - msg.timestamp().count();
247 }
248
249 std::ostringstream csv_line;
250 csv_line << timestamp_us << ","
251 << (msg.message_type() == Message::Type::INTEREST ? "INTEREST" : "RESPONSE") << ","
252 << direction << ","
253 << (direction == "SEND" ? _address.to_string() : msg.origin().to_string()) << ","
254 << (direction == "SEND" ? "BROADCAST" : _address.to_string()) << ","
255 << msg.unit() << ","
256 << msg.period().count() << ","
257 << msg.value_size() << ","
258 << latency_us;
259
260 _csv_logger->log(csv_line.str());
261}
