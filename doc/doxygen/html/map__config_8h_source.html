102inline void MapConfig::parse_config_file(const std::string& file_path) {
103 std::ifstream file(file_path);
104 if (!file.is_open()) {
105 throw std::runtime_error("Cannot open configuration file: " + file_path);
106 }
107
108 std::string content((std::istreambuf_iterator<char>(file)),
109 std::istreambuf_iterator<char>());
110
111 // Try to parse RSU configurations - first try multiple RSUs, then single RSU
112 if (content.find("\"rsus\":") != std::string::npos) {
113 // Multiple RSUs configuration (array)
114 parse_multiple_rsus(content);
115 } else if (content.find("\"rsu\":") != std::string::npos) {
116 // Single RSU configuration (object)
117 try {
118 std::string rsu_section = extract_json_section(content, "rsu");
119 parse_single_rsu(rsu_section);
120 } catch (const std::exception& e) {
121 throw std::runtime_error("Error parsing single RSU section: " + std::string(e.what()));
122 }
123 } else {
124 throw std::runtime_error("Neither 'rsu' nor 'rsus' section found in config file");
125 }
126
127 // Parse vehicle config section
128 std::string vehicle_section = extract_json_section(content, "vehicles");
129 _vehicle_config.default_count = extract_uint_value(vehicle_section, "default_count");
130 _vehicle_config.speed_kmh = extract_double_value(vehicle_section, "speed_kmh");
131
132 // Parse simulation config section
133 std::string simulation_section = extract_json_section(content, "simulation");
134 _simulation.duration_s = extract_uint_value(simulation_section, "duration_s");
135 _simulation.update_interval_ms = extract_uint_value(simulation_section, "update_interval_ms");
136 _simulation.default_transmission_radius_m = extract_double_value(simulation_section, "default_transmission_radius_m");
137
138 // trajectory_generator_script is optional, provide default if not found
139 try {
140 _simulation.trajectory_generator_script = extract_string_value(simulation_section, "trajectory_generator_script");
141 } catch (const std::exception&) {
142 _simulation.trajectory_generator_script = "scripts/trajectory_generator_map_1.py"; // Default fallback
143 }
144
145 // Parse logging config section
146 std::string logging_section = extract_json_section(content, "logging");
147 _logging.trajectory_dir = extract_string_value(logging_section, "trajectory_dir");
148
149 // Parse waypoints and routes
150 parse_waypoints(content);
151 parse_routes(content);
152}

154inline void MapConfig::parse_waypoints(const std::string& content) {
155 // Simple parser for waypoints array
156 size_t waypoints_start = content.find("\"waypoints\":");
157 if (waypoints_start == std::string::npos) return;
158
159 size_t array_start = content.find("[", waypoints_start);
160 size_t array_end = content.find("]", array_start);
161
162 std::string waypoints_section = content.substr(array_start + 1, array_end - array_start - 1);
163
164 // Parse each waypoint object
165 size_t pos = 0;
166 while (pos < waypoints_section.length()) {
167 size_t obj_start = waypoints_section.find("{", pos);
168 if (obj_start == std::string::npos) break;
169
170 size_t obj_end = waypoints_section.find("}", obj_start);
171 if (obj_end == std::string::npos) break;
172
173 std::string waypoint_obj = waypoints_section.substr(obj_start, obj_end - obj_start + 1);
174
175 Waypoint wp;
176 wp.name = extract_string_value(waypoint_obj, "name");
177 wp.x = extract_double_value(waypoint_obj, "x");
178 wp.y = extract_double_value(waypoint_obj, "y");
179
180 _waypoints.push_back(wp);
181 pos = obj_end + 1;
182 }
183}

193 std::string routes_section = content.substr(array_start + 1, array_end - array_start - 1);
194
195 // Parse each route object
196 size_t pos = 0;
197 while (pos < routes_section.length()) {
198 size_t obj_start = routes_section.find("{", pos);
199 if (obj_start == std::string::npos) break;
200
201 size_t obj_end = routes_section.find("}", obj_start);
202 if (obj_end == std::string::npos) break;
203
204 std::string route_obj = routes_section.substr(obj_start, obj_end - obj_start + 1);
205
206 Route route;
207 route.name = extract_string_value(route_obj, "name");
208
209 // Parse waypoints array within route
210 size_t wp_array_start = route_obj.find("[");
211 size_t wp_array_end = route_obj.find("]", wp_array_start);
212
213 if (wp_array_start != std::string::npos && wp_array_end != std::string::npos) {
214 std::string wp_list = route_obj.substr(wp_array_start + 1, wp_array_end - wp_array_start - 1);
215
216 // Simple comma-separated parsing
217 size_t wp_pos = 0;
218 while (wp_pos < wp_list.length()) {
219 size_t quote_start = wp_list.find("\"", wp_pos);
220 if (quote_start == std::string::npos) break;
221
222 size_t quote_end = wp_list.find("\"", quote_start + 1);
223 if (quote_end == std::string::npos) break;
224
225 std::string waypoint_name = wp_list.substr(quote_start + 1, quote_end - quote_start - 1);
226 route.waypoint_names.push_back(waypoint_name);

228 wp_pos = quote_end + 1;
229 }
230 }

248inline void MapConfig::parse_multiple_rsus(const std::string& content) {
249 // Simple parser for rsus array
250 size_t rsus_start = content.find("\"rsus\":");
251 if (rsus_start == std::string::npos) return;
252
253 size_t array_start = content.find("[", rsus_start);
254 size_t array_end = content.find("]", array_start);
255
256 if (array_start == std::string::npos || array_end == std::string::npos) return;
257
258 std::string rsus_section = content.substr(array_start + 1, array_end - array_start - 1);
259
260 // Parse each RSU object
261 size_t pos = 0;
262 while (pos < rsus_section.length()) {
263 size_t obj_start = rsus_section.find("{", pos);
264 if (obj_start == std::string::npos) break;
265
266 // Find matching closing brace using proper brace counting
267 size_t obj_end = obj_start;
268 int brace_count = 0;
269 while (obj_end < rsus_section.length()) {
270 if (rsus_section[obj_end] == '{') {
271 brace_count++;
272 } else if (rsus_section[obj_end] == '}') {
273 brace_count--;
274 if (brace_count == 0) {
275 break;
276 }
277 }
278 obj_end++;
279 }
280
281 if (brace_count != 0 || obj_end >= rsus_section.length()) {
282 break; // Malformed JSON
283 }
284
285 std::string rsu_obj = rsus_section.substr(obj_start, obj_end - obj_start + 1);
286
287 RSUConfig rsu_config;
288 rsu_config.id = extract_uint_value(rsu_obj, "id");
289 rsu_config.unit = extract_uint_value(rsu_obj, "unit");
290 rsu_config.broadcast_period = std::chrono::milliseconds(extract_uint_value(rsu_obj, "broadcast_period_ms"));
291
292 // Parse RSU position subsection
293 std::string rsu_position_section = extract_json_section(rsu_obj, "position");
294 rsu_config.x = extract_double_value(rsu_position_section, "x");
295 rsu_config.y = extract_double_value(rsu_position_section, "y");
296
297 _rsu_configs.push_back(rsu_config);
298 pos = obj_end + 1;
299 }

302inline std::string MapConfig::extract_json_section(const std::string& content, const std::string& section_name) const {
303 std::string search_pattern = "\"" + section_name + "\":";
304 size_t pos = content.find(search_pattern);
305 if (pos == std::string::npos) {
306 throw std::runtime_error("Section not found in config: " + section_name);
307 }
308
309 pos += search_pattern.length();
310
311 // Skip whitespace and find opening brace
312 while (pos < content.length() && std::isspace(content[pos])) {
313 pos++;
314 }
315
316 if (pos >= content.length() || content[pos] != '{') {
317 throw std::runtime_error("Invalid JSON structure for section: " + section_name);
318 }
319
320 // Find matching closing brace
321 size_t start_pos = pos;
322 int brace_count = 0;
323 while (pos < content.length()) {
324 if (content[pos] == '{') {
325 brace_count++;
326 } else if (content[pos] == '}') {
327 brace_count--;
328 if (brace_count == 0) {
329 break;
330 }
331 }
332 pos++;
333 }
334
335 if (brace_count != 0) {
336 throw std::runtime_error("Unmatched braces in section: " + section_name);
337 }
338
339 return content.substr(start_pos, pos - start_pos + 1);
340}

342inline double MapConfig::extract_double_value(const std::string& content, const std::string& key) const {
343 std::string search_pattern = "\"" + key + "\":";
344 size_t pos = content.find(search_pattern);
345 if (pos == std::string::npos) {
346 throw std::runtime_error("Key not found in config: " + key);
347 }
348
349 pos += search_pattern.length();
350
351 // Skip whitespace
352 while (pos < content.length() && std::isspace(content[pos])) {
353 pos++;
354 }
355
356 // Extract the numeric value
357 size_t end_pos = pos;
358 while (end_pos < content.length() &&
359 (std::isdigit(content[end_pos]) || content[end_pos] == '.' || content[end_pos] == '-')) {
360 end_pos++;
361 }
362
363 if (pos == end_pos) {
364 throw std::runtime_error("Invalid numeric value for key: " + key);
365 }
366
367 return std::stod(content.substr(pos, end_pos - pos));
368}

374inline std::string MapConfig::extract_string_value(const std::string& content, const std::string& key) const {
375 std::string search_pattern = "\"" + key + "\":";
376 size_t pos = content.find(search_pattern);
377 if (pos == std::string::npos) {
378 throw std::runtime_error("Key not found in config: " + key);
379 }
380
381 pos += search_pattern.length();
382
383 // Skip whitespace and find opening quote
384 while (pos < content.length() && (std::isspace(content[pos]) || content[pos] == '"')) {
385 if (content[pos] == '"') {
386 pos++;
387 break;
388 }
389 pos++;
390 }
391
392 // Find closing quote
393 size_t end_pos = pos;
394 while (end_pos < content.length() && content[end_pos] != '"') {
395 end_pos++;
396 }
397
398 if (end_pos == content.length()) {
399 throw std::runtime_error("Unterminated string value for key: " + key);
400 }
401
402 return content.substr(pos, end_pos - pos);
403}
